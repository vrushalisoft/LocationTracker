$('#date').on('change', function() { var newVal = $(this).val().split('-'), //renamed new_val to newVal, always stick to one naming convention dateParts = { year: parseInt(newVal[0], 10), month: parseInt(newVal[1], 10), day: parseInt(newVal[2], 10) }; $('#anchor').append($.map(dateParts, function (index, key) { var name = 'date_' + key; return $('', { type: 'hidden', name: name, id: name, value: dateParts[key] }); })); });
